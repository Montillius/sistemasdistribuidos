import socket
import threading

# configuracoes do servidor
HOST = 'localhost'
PORT = 12345

# cria o socket
server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server_socket.bind((HOST, PORT))
server_socket.listen()

# funcao que sera executada em uma thread separada para cada cliente
def handle_client(client_socket):
  # recebe o resultado da raiz quadrada do cliente
  result = client_socket.recv(1024).decode('utf-8')

  # imprime o resultado
  print(f'Resultado recebido do cliente: {result}')

  # fecha a conexao com o cliente
  client_socket.close()

# loop principal do servidor
while True:
  # aceita uma nova conexao
  client_socket, addr = server_socket.accept()

  # cria uma nova thread para lidar com a conexao do cliente
  client_thread = threading.Thread(target=handle_client, args=(client_socket,))
  client_thread.start()
