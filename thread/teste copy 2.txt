import socket
import threading

# Cria um socket TCP/IP
sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

# Conecta o socket ao servidor na porta especificada
server_address = ('localhost', 10000)
sock.connect(server_address)

# Função para calcular a raiz quadrada em uma thread separada
def calc_sqrt(n):
    result = n ** 0.5
    sock.sendall(str(result).encode())

# Envia o número para o servidor
number = 5
sock.sendall(str(number).encode())

# Inicia duas threads para calcular a raiz quadrada do número
thread1 = threading.Thread(target=calc_sqrt, args=(number,))
thread2 = threading.Thread(target=calc_sqrt, args=(number,))
thread1.start()
thread2.start()

# Fecha o socket depois que as threads terminarem
thread1.join()
thread2.join()
sock.close()
